Package graph implements data structure for directed or undirected graphs.

The reference implementation uses adjacency list representation: https://en.wikipedia.org/wiki/Adjacency_list

▹ Example

▾ Example

Code:

package graph_test

import (
    "fmt"
    "sort"

    "hsecode.com/stdlib/graph"
)

type Int int

func (v Int) ID() int {
    return int(v)
}

func Example() {
    G := graph.New(graph.Undirected)

    // Create the following graph with no edge data:
    //
    //  1---2 \
    //  | / |  3
    //  5---4 /

    for i := 1; i <= 5; i++ {
        G.AddNode(Int(i))
    }
    G.AddEdge(1, 2, nil)
    G.AddEdge(2, 3, nil)
    G.AddEdge(3, 4, nil)
    G.AddEdge(4, 5, nil)
    G.AddEdge(5, 1, nil)
    G.AddEdge(2, 4, nil)
    G.AddEdge(2, 5, nil)

    // Compute neighbour nodes for 2
    nodes := make([]int, 0)
    G.Neighbours(2, func(v graph.Node, e interface{}) {
        nodes = append(nodes, v.ID())
    })

    sort.Ints(nodes)
    fmt.Println(nodes)
    // Output: [1 3 4 5]
}

func (*Graph) AddEdge ¶ 

func (g *Graph) AddEdge(u, v int, edgeData interface{})

AddEdge creates an edge between nodes u and v augmented with the provided data. If the edge between u and v already exists, the method overwrites the edge data. Panics if the given nodes do not exist.

func (*Graph) AddNode ¶ 

func (g *Graph) AddNode(node Node)

AddNode creates a new node or overwrites an existing one with the same ID. In the latter case, the method preserves edges incident to the node.

▹ Example (Overwrite)

▾ Example (Overwrite)

Code:

package graph_test

import (
    "fmt"
    "hsecode.com/stdlib/graph"
)

type Person struct {
    Id   int
    Name string
}

func (d Person) ID() int { return d.Id }

func ExampleGraph_AddNode_overwrite() {
    G := graph.New(graph.Undirected)

    G.AddNode(Person{Id: 1, Name: "Fred Weasley"})
    node, _ := G.Node(1)

    fmt.Println(node.(Person).Name)

    // data will be overwritten due to the same ID
    G.AddNode(Person{Id: 1, Name: "George Weasley"})

    node, _ = G.Node(1)
    fmt.Println(node.(Person).Name)

    // Output:
    // Fred Weasley
    // George Weasley
}

func (*Graph) Edges ¶ 

func (g *Graph) Edges(f func(u, v Node, edgeData interface{}))

Edges applies the provided function f to all graph edges in arbitrary order.

NOTE: for undirected graphs the function is called once for every connected pair of nodes. For example, for the graph (2)--(3) the function must be called once either with (u=2,v=3) or (u=3,v=2).

▹ Example (Directed)

▾ Example (Directed)

Code:

g := graph.New(graph.Directed)
g.AddNode(Int(2))
g.AddNode(Int(3))

g.AddEdge(2, 3, "forward edge")
g.AddEdge(3, 2, "backward edge")

g.Edges(func(u, v graph.Node, e interface{}) {
    fmt.Println(u, v, e)
})

Output:

2 3 forward edge
3 2 backward edge
